#include<iostream>
using namespace std;
int a[10]={1,3,4,2,6,5,9,8,7,0};
void BubbleSort(int *a,int n)
{
	for(int i=0;i<n-1;i++)
	{
		for(int j=0;j<n-i-1;j++)
		{
			if(a[j]>a[j+1])
			swap(a[j],a[j+1]);
		}
	}
}
void SelectSort(int *a,int n)
{
	for(int i=0;i<n;i++)
	{
		int k=i;
		for(int j=i;j<n;j++)
		{
			if(a[j]<a[k])
			k=j; 
		}
		swap(a[i],a[k]);
	}
}
void InsertSort(int *a,int n)
{
	for(int i=1;i<n;i++)
	{
		int j=i,temp=a[i];
		while(j>0&&temp<a[j-1])
		{
			a[j]=a[j-1];
			j--;
		}
		a[j]=temp;
	}
}
